1. Main flow:
   ```typescript
   async function main() {
     const steps: SeedStep[] = [];
     try {
       // Load config and init context
       const config = await loadConfig();
       const context = await initContext(config);
       const tagId = await ensureMarkerTag(context);
       
       // Create RFP
       const rfpId = await createRfp(context);
       steps.push({ name: 'Create RFP', status: 'done' });
       
       // Create 2+ documents
       const doc1Id = await createDocument(context, rfpId);
       const doc2Id = await createDocument(context, rfpId);
       steps.push({ name: 'Attach documents (2)', status: 'done' });
       
       // Lookup or seed vendor (reuse existing UAT vendor if available)
       const vendorPartyId = await resolveVendor(context); // env var or fixture lookup
       const inviteId = await inviteVendor(context, rfpId, vendorPartyId);
       steps.push({ name: 'Invite vendor', status: 'done' });
       
       // Create bid from vendor
       const bidId = await createBid(context, rfpId, vendorPartyId);
       steps.push({ name: 'Submit bid', status: 'done' });
       
       // Create form submission on bid (per Phase 16)
       const formSubmissionId = await createFormSubmission(context, bidId);
       steps.push({ name: 'Submit form responses', status: 'done' });
       
       // Create pilot project linked to RFP
       const pilotId = await createPilot(context, rfpId);
       steps.push({ name: 'Create pilot project', status: 'done' });
       
       // Print summary and exit 0
       printSummary({ rfpId, documentIds: [doc1Id, doc2Id], vendorPartyId, bidId, formSubmissionId, pilotId }, steps);
       process.exit(0);
     } catch (err) {
       // On any error, print what was created + error + exit 1
       console.error(`\n❌ Seed failed at step: ${JSON.stringify(steps[steps.length - 1])}`);
       console.error(`Error: ${(err as Error).message}`);
       process.exit(1);
     }
   }
   
   main();
   ```

1. Main flow:
   ```typescript
   async function main() {
     try {
       const config = await loadConfig();
       const context = await initContext(config);
       const entities = await cleanupByMarkerTag(context);
       
       if (entities.length === 0) {
         console.info('No demo data to clean up.');
         process.exit(0);
       }
       
       // Delete in reverse-dependency order
       const deleted = await deleteInOrder(context, entities);
       
       console.info(`\n✓ Cleanup complete. ${deleted.count} resources removed.`);
       process.exit(0);
     } catch (err) {
       console.error(`\n❌ Cleanup failed: ${(err as Error).message}`);
       process.exit(1);
     }
   }
   ```
