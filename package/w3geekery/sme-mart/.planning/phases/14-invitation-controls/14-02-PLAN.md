import { Component, OnInit, inject, signal } from '@angular/core';
import { CommonModule } from '@angular/common';
import { MatCardModule } from '@angular/material/card';
import { MatButtonModule } from '@angular/material/button';
import { MatChipsModule } from '@angular/material/chips';
import { MatIconModule } from '@angular/material/icon';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';
import { MatDividerModule } from '@angular/material/divider';
import { MatTabsModule } from '@angular/material/tabs';
import { MatDialog, MatDialogModule } from '@angular/material/dialog';
import { ZbEmptyStateContainerComponent, ZbResourceStatusComponent } from '@zerobias-org/ngx-library';
import { RfpInvitationService } from '@core/services/rfp-invitation.service';
import { SmeMartProjectService } from '@core/services/sme-mart-project.service';
import { ZerobiasClientService } from '@core/services/zerobias-client.service';
import type { RfpInvitation } from '@core/models';

@Component({
  selector: 'app-my-invitations',
  standalone: true,
  imports: [
    CommonModule,
    MatCardModule,
    MatButtonModule,
    MatChipsModule,
    MatIconModule,
    MatProgressSpinnerModule,
    MatDividerModule,
    MatTabsModule,
    MatDialogModule,
    ZbEmptyStateContainerComponent,
    ZbResourceStatusComponent,
  ],
  templateUrl: './my-invitations.component.html',
  styleUrls: ['./my-invitations.component.scss'],
})
export class MyInvitationsComponent implements OnInit {
  private readonly rfpInvitation = inject(RfpInvitationService);
  private readonly smeMartProject = inject(SmeMartProjectService);
  private readonly zerobiasClient = inject(ZerobiasClientService);
  private readonly dialog = inject(MatDialog);

  invitations = signal<RfpInvitation[]>([]);
  projects = signal<Map<string, any>>(new Map());
  loading = signal(true);
  error = signal<string | null>(null);

  selectedFilter = signal<'all' | 'pending' | 'accepted'>('all');

  ngOnInit(): void {
    this.loadInvitations();
  }

  private async loadInvitations(): Promise<void> {
    this.loading.set(true);
    this.error.set(null);

    try {
      const currentOrg = await this.zerobiasClient.getCurrentOrg();
      const invites = await this.rfpInvitation.listByVendorOrg(currentOrg.id);
      this.invitations.set(invites);

      // Preload projects for each invitation
      const projectIds = new Set(invites.map(i => i.projectId));
      for (const projectId of projectIds) {
        const project = await this.smeMartProject.getProject(projectId);
        const map = this.projects();
        map.set(projectId, project);
      }
    } catch (err) {
      this.error.set((err as Error).message);
    } finally {
      this.loading.set(false);
    }
  }

  get filteredInvitations(): RfpInvitation[] {
    const filter = this.selectedFilter();
    const all = this.invitations();

    if (filter === 'all') return all;
    return all.filter(i => i.status === filter);
  }

  getProjectName(projectId: string): string {
    const project = this.projects().get(projectId);
    return project?.name ?? 'Unknown RFP';
  }

  getStatusLabel(status: string): string {
    const labels: Record<string, string> = {
      pending: 'Pending Review',
      accepted: 'Accepted',
      declined: 'Declined',
      requested: 'Request Pending',
    };
    return labels[status] ?? status;
  }

  async acceptInvitation(invitationId: string): Promise<void> {
    try {
      await this.rfpInvitation.acceptInvitation(invitationId);
      await this.loadInvitations();
    } catch (err) {
      this.error.set((err as Error).message);
    }
  }

  async declineInvitation(invitationId: string): Promise<void> {
    const dialogRef = this.dialog.open(DeclineReasonDialogComponent);
    dialogRef.afterClosed().subscribe(async (reason?: string) => {
      if (reason !== undefined) {
        try {
          await this.rfpInvitation.declineInvitation(invitationId, reason);
          await this.loadInvitations();
        } catch (err) {
          this.error.set((err as Error).message);
        }
      }
    });
  }

  navigateToRfp(projectId: string): void {
    // Navigate to project detail page
    // Use router to navigate to /projects/:projectId
  }
}

// Decline reason dialog component
@Component({
  selector: 'app-decline-reason-dialog',
  standalone: true,
  imports: [CommonModule, MatDialogModule, MatButtonModule],
  template: `
    <h2 mat-dialog-title>Decline Invitation</h2>
    <mat-dialog-content>
      <p>Would you like to provide a reason for declining?</p>
      <textarea [(ngModel)]="reason" placeholder="Optional reason..."></textarea>
    </mat-dialog-content>
    <mat-dialog-actions align="end">
      <button mat-button (click)="onCancel()">Cancel</button>
      <button mat-raised-button color="primary" (click)="onConfirm()">Decline</button>
    </mat-dialog-actions>
  `,
})
export class DeclineReasonDialogComponent {
  reason = '';

  constructor(public dialogRef: MatDialogRef<DeclineReasonDialogComponent>) {}

  onCancel(): void {
    this.dialogRef.close();
  }

  onConfirm(): void {
    this.dialogRef.close(this.reason);
  }
}
