```typescript
async submitBid(): Promise<void> {
  // Check if form is required (project has formConfig)
  const project = await this.projectService.getById(this.projectId);
  if (project?.formConfig) {
    // Form is required - check if submitted
    const submission = await this.formSubmissionService.getByProjectAndBid(
      this.projectId,
      this.currentBidId
    );
    
    if (!submission || submission.status === 'draft') {
      this.showError('Form submission required before submitting bid');
      return;
    }
  }

  // Proceed with bid submission
  await this.bidsService.createOrUpdate(...);
}
```
