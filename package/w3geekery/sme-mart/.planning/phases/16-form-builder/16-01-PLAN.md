```typescript
import { Injectable, inject } from '@angular/core';
import { UUID } from '@zerobias-org/types-core-js';
import { Observable } from 'rxjs';
import { FormSubmission, FormBuilderConfig } from '../models/form-builder.model';
import { PipelineWriteService } from './pipeline-write.service';
import { GraphQLReadService } from './graphql-read.service';

@Injectable({ providedIn: 'root' })
export class FormSubmissionService {
  private pipelineWrite = inject(PipelineWriteService);
  private gqlRead = inject(GraphQLReadService);

  /**
   * Create a new form submission (draft) for a project/bid
   */
  async create(projectId: UUID, bidId: UUID): Promise<FormSubmission> {
    const submission: FormSubmission = {
      id: crypto.randomUUID(),
      projectId,
      bidId,
      submissionData: {},
      status: 'draft',
      createdAt: new Date(),
      updatedAt: new Date(),
    };

    // Persist via Pipeline
    // Note: Update SME_MART_CLASS_IDS.FORM_SUBMISSION with class ID from Plan 00 Task 3
    await this.pipelineWrite.receive(
      SME_MART_PIPELINE_ID,
      [{
        classId: this.getFormSubmissionClassId(),
        object: {
          newFormSubmission: {
            id: submission.id,
            projectId: submission.projectId,
            bidId: submission.bidId,
            submissionData: JSON.stringify(submission.submissionData),
            status: submission.status,
            createdAt: submission.createdAt,
            updatedAt: submission.updatedAt,
          }
        }
      }]
    );

    return submission;
  }

  /**
   * Fetch FormSubmission by id
   */
  async getById(id: UUID): Promise<FormSubmission | null> {
    // GQL query for FormSubmission by id
    const result = await this.gqlRead.query({
      query: gql`query GetFormSubmission($id: UUID!) {
        formSubmission(id: $id) {
          id
          projectId
          bidId
          submissionData
          status
          submittedAt
          reviewedAt
          reviewedBy
          createdAt
          updatedAt
        }
      }`,
      variables: { id },
    });

    return result?.data?.formSubmission || null;
  }

  /**
   * Get FormSubmission for a specific project/bid pair
   */
  async getByProjectAndBid(projectId: UUID, bidId: UUID): Promise<FormSubmission | null> {
    const result = await this.gqlRead.query({
      query: gql`query GetFormSubmissionByProjectBid($projectId: UUID!, $bidId: UUID!) {
        formSubmissions(filter: { projectId: $projectId, bidId: $bidId }, pageSize: 1) {
          items {
            id
            projectId
            bidId
            submissionData
            status
            submittedAt
            reviewedAt
            reviewedBy
            createdAt
            updatedAt
          }
        }
      }`,
      variables: { projectId, bidId },
    });

    return result?.data?.formSubmissions?.items?.[0] || null;
  }

  /**
   * Update FormSubmission with new data and status
   */
  async update(id: UUID, updates: Partial<FormSubmission>): Promise<FormSubmission> {
    // Pipeline update call
    await this.pipelineWrite.receive(
      SME_MART_PIPELINE_ID,
      [{
        classId: this.getFormSubmissionClassId(),
        object: {
          id,
          ...Object.keys(updates).reduce((acc, key) => {
            if (key === 'submissionData' && typeof updates[key] === 'object') {
              acc[key] = JSON.stringify(updates[key]);
            } else {
              acc[key] = updates[key];
            }
            return acc;
          }, {})
        }
      }]
    );

    return this.getById(id) as Promise<FormSubmission>;
  }

  /**
   * Mark submission as reviewed by a buyer
   */
  async markReviewed(id: UUID, reviewedBy: UUID): Promise<FormSubmission> {
    return this.update(id, {
      status: 'reviewed',
      reviewedAt: new Date(),
      reviewedBy,
    });
  }

  /**
   * Check if submission is in draft state
   */
  isDraft(submission: FormSubmission): boolean {
    return submission.status === 'draft';
  }

  /**
   * Check if form is locked (any submission exists for this project)
   * Returns true if form should be read-only, false if buyer can still edit form config
   */
  async getFormSubmissionLock(projectId: UUID): Promise<boolean> {
    const result = await this.gqlRead.query({
      query: gql`query CheckFormSubmissions($projectId: UUID!) {
        formSubmissions(filter: { projectId: $projectId }, pageSize: 1) {
          total
        }
      }`,
      variables: { projectId },
    });

    return (result?.data?.formSubmissions?.total || 0) > 0;
  }

  /**
   * List all submissions for a project
   */
  async listByProject(projectId: UUID): Promise<FormSubmission[]> {
    const result = await this.gqlRead.query({
      query: gql`query ListFormSubmissions($projectId: UUID!) {
        formSubmissions(filter: { projectId: $projectId }) {
          items {
            id
            projectId
            bidId
            submissionData
            status
            submittedAt
            reviewedAt
            reviewedBy
            createdAt
            updatedAt
          }
        }
      }`,
      variables: { projectId },
    });

    return result?.data?.formSubmissions?.items || [];
  }

  private getFormSubmissionClassId(): string {
    // Update this UUID with the actual FormSubmission class ID from Plan 00 Task 3
    return 'PLACEHOLDER_UPDATE_WITH_CLASS_ID_FROM_PLAN_00';
  }
}
```

```typescript
import {
  Component, input, output, signal, computed, inject,
  ChangeDetectionStrategy, OnInit
} from '@angular/core';
import {
  ReactiveFormsModule, FormBuilder, FormGroup, FormArray, Validators
} from '@angular/forms';
import { CommonModule } from '@angular/common';
import { MatExpansionModule } from '@angular/material/expansion';
import { MatFormFieldModule } from '@angular/material/form-field';
import { MatInputModule } from '@angular/material/input';
import { MatSelectModule } from '@angular/material/select';
import { MatButtonModule } from '@angular/material/button';
import { MatIconModule } from '@angular/material/icon';
import { MatDragDropModule } from '@angular/cdk/drag-drop';
import { CdkDragDrop } from '@angular/cdk/drag-drop';
import { DynamicFormRendererComponent } from './dynamic-form-renderer.component';
import { FormFieldEditorComponent } from './form-field-editor.component';
import { FormBuilderConfig, FormFieldConfig, FORM_FIELD_TYPES } from '../../models/form-builder.model';

@Component({
  selector: 'app-form-builder',
  standalone: true,
  imports: [
    CommonModule,
    ReactiveFormsModule,
    MatExpansionModule,
    MatFormFieldModule,
    MatInputModule,
    MatSelectModule,
    MatButtonModule,
    MatIconModule,
    MatDragDropModule,
    DynamicFormRendererComponent,
    FormFieldEditorComponent,
  ],
  templateUrl: './form-builder.component.html',
  styleUrls: ['./form-builder.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class FormBuilderComponent implements OnInit {
  formConfig = input<FormBuilderConfig>({
    version: 1,
    fields: [],
  });
  isLocked = input<boolean>(false);
  formConfigChange = output<FormBuilderConfig>();

  private fb = inject(FormBuilder);

  formGroup = signal<FormGroup | null>(null);
  showPreview = signal(false);
  fieldWarning = signal(false);

  ngOnInit(): void {
    const config = this.formConfig();
    const formArray = this.fb.array(
      config.fields.map(field => this.createFieldFormGroup(field))
    );
    this.formGroup.set(this.fb.group({ fields: formArray }));
  }

  private createFieldFormGroup(field: FormFieldConfig): FormGroup {
    return this.fb.group({
      id: [field.id || crypto.randomUUID()],
      type: [field.type, Validators.required],
      label: [field.label, Validators.required],
      required: [field.required || false],
      placeholder: [field.placeholder || ''],
      description: [field.description || ''],
      textValidation: [field.textValidation || {}],
      dropdownOptions: [field.dropdownOptions || []],
      numberValidation: [field.numberValidation || {}],
      fileUploadConfig: [field.fileUploadConfig || {}],
      sectionId: [field.sectionId || null],
    });
  }

  addField(): void {
    const fieldsArray = this.getFieldsArray();
    if (!fieldsArray) return;

    const newField = this.createFieldFormGroup({
      id: crypto.randomUUID(),
      type: 'text',
      label: 'New Field',
      required: false,
    });
    fieldsArray.push(newField);
    this.checkFieldWarning();
    this.emitChange();
  }

  removeField(index: number): void {
    const fieldsArray = this.getFieldsArray();
    if (!fieldsArray) return;
    fieldsArray.removeAt(index);
    this.checkFieldWarning();
    this.emitChange();
  }

  drop(event: CdkDragDrop<any>): void {
    const fieldsArray = this.getFieldsArray();
    if (!fieldsArray) return;

    const previousIndex = event.previousIndex;
    const currentIndex = event.currentIndex;

    if (previousIndex !== currentIndex) {
      const item = fieldsArray.at(previousIndex);
      fieldsArray.removeAt(previousIndex);
      fieldsArray.insert(currentIndex, item);
      this.emitChange();
    }
  }

  private getFieldsArray(): FormArray | null {
    return this.formGroup()?.get('fields') as FormArray | null;
  }

  private checkFieldWarning(): void {
    const count = this.getFieldsArray()?.length || 0;
    this.fieldWarning.set(count >= 25);
  }

  private emitChange(): void {
    const fieldsArray = this.getFieldsArray();
    if (!fieldsArray) return;

    const newConfig: FormBuilderConfig = {
      version: 1,
      fields: fieldsArray.value.map((f: any) => ({
        id: f.id,
        type: f.type,
        label: f.label,
        required: f.required,
        placeholder: f.placeholder,
        description: f.description,
        textValidation: f.textValidation,
        dropdownOptions: f.dropdownOptions,
        numberValidation: f.numberValidation,
        fileUploadConfig: f.fileUploadConfig,
        sectionId: f.sectionId,
      } as FormFieldConfig)),
    };
    this.formConfigChange.emit(newConfig);
  }

  togglePreview(): void {
    this.showPreview.update(v => !v);
  }

  getFormFieldTypes(): typeof FORM_FIELD_TYPES {
    return FORM_FIELD_TYPES;
  }
}
```

```typescript
import {
  Component, input, output, signal, inject,
  ChangeDetectionStrategy, OnInit
} from '@angular/core';
import {
  ReactiveFormsModule, FormBuilder, FormGroup, Validators, FormArray
} from '@angular/forms';
import { CommonModule } from '@angular/common';
import { MatFormFieldModule } from '@angular/material/form-field';
import { MatInputModule } from '@angular/material/input';
import { MatSelectModule } from '@angular/material/select';
import { MatCheckboxModule } from '@angular/material/checkbox';
import { MatButtonModule } from '@angular/material/button';
import { MatIconModule } from '@angular/material/icon';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';
import {
  FormBuilderConfig, FormFieldConfig, FormSubmission, FormFieldType
} from '../../models/form-builder.model';
import { ZerobiasClientAppService } from '@zerobias-com/zerobias-angular-client';

type FormMode = 'preview' | 'fill' | 'review';

@Component({
  selector: 'app-dynamic-form-renderer',
  standalone: true,
  imports: [
    CommonModule,
    ReactiveFormsModule,
    MatFormFieldModule,
    MatInputModule,
    MatSelectModule,
    MatCheckboxModule,
    MatButtonModule,
    MatIconModule,
    MatProgressSpinnerModule,
  ],

  styleUrls: ['./dynamic-form-renderer.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class DynamicFormRendererComponent implements OnInit {
  config = input<FormBuilderConfig>({ version: 1, fields: [] });
  mode = input<FormMode>('preview');
  submission = input<Partial<FormSubmission>>({ submissionData: {} });

  submit = output<Record<string, unknown>>();
  draftSave = output<Record<string, unknown>>();

  private fb = inject(FormBuilder);
  private zb = inject(ZerobiasClientAppService);

  formGroup = signal<FormGroup | null>(null);
  isSubmitting = signal(false);
  fileUploadInProgress = signal<Record<string, boolean>>({});

  ngOnInit(): void {
    const config = this.config();
    const group: Record<string, any> = {};

    config.fields.forEach(field => {
      const initialValue = this.getInitialValue(field);
      const validators = this.getValidators(field);
      group[field.id] = [initialValue, validators];
    });

    this.formGroup.set(this.fb.group(group));
  }

  private getInitialValue(field: FormFieldConfig): any {
    const submissionData = this.submission().submissionData || {};
    const value = submissionData[field.id];

    if (this.mode() === 'preview') {
      return this.getSampleValue(field.type);
    }

    return value || (field.type === 'checkbox' ? false : '');
  }

  private getSampleValue(type: FormFieldType): any {
    switch (type) {
      case 'text':
        return 'Sample text';
      case 'textarea':
        return 'Sample text area content';
      case 'dropdown':
        return 'option1';
      case 'number':
        return 42;
      case 'file':
        return null;
      case 'checkbox':
        return false;
    }
  }

  private getValidators(field: FormFieldConfig): any[] {
    const validators = [];

    if (field.required) {
      validators.push(Validators.required);
    }

    if (field.type === 'text' && field.textValidation) {
      if (field.textValidation.minLength) {
        validators.push(Validators.minLength(field.textValidation.minLength));
      }
      if (field.textValidation.maxLength) {
        validators.push(Validators.maxLength(field.textValidation.maxLength));
      }
      if (field.textValidation.pattern === 'email') {
        validators.push(Validators.email);
      } else if (field.textValidation.pattern === 'url') {
        validators.push(Validators.pattern(/^https?:\/\/.+/));
      } else if (field.textValidation.pattern === 'phone') {
        validators.push(Validators.pattern(/^\d{10}$/));
      } else if (field.textValidation.pattern === 'custom' && field.textValidation.patternValue) {
        validators.push(Validators.pattern(field.textValidation.patternValue));
      }
    }

    if (field.type === 'textarea' && field.textValidation) {
      if (field.textValidation.minLength) {
        validators.push(Validators.minLength(field.textValidation.minLength));
      }
      if (field.textValidation.maxLength) {
        validators.push(Validators.maxLength(field.textValidation.maxLength));
      }
    }

    if (field.type === 'number' && field.numberValidation) {
      if (field.numberValidation.min !== undefined) {
        validators.push(Validators.min(field.numberValidation.min));
      }
      if (field.numberValidation.max !== undefined) {
        validators.push(Validators.max(field.numberValidation.max));
      }
    }

    return validators;
  }

  onSubmit(): void {
    const form = this.formGroup();
    if (!form || !form.valid) {
      return;
    }

    this.isSubmitting.set(true);
    this.submit.emit(form.value);
  }

  onDraftSave(): void {
    const form = this.formGroup();
    if (!form) {
      return;
    }
    this.draftSave.emit(form.value);
  }

  async onFileUpload(fieldId: string, event: Event): Promise<void> {
    const input = event.target as HTMLInputElement;
    const file = input.files?.[0];
    if (!file) return;

    // Mark upload in progress
    this.fileUploadInProgress.update(state => ({ ...state, [fieldId]: true }));

    try {
      // Use ZB FileService to upload
      const fileService = this.zb.getClientApi().fileClient;
      const uploadedFile = await fileService.uploadFile(file);

      // Store file reference in form
      const form = this.formGroup();
      if (form && uploadedFile?.id) {
        form.get(fieldId)?.setValue({
          fileId: uploadedFile.id,
          fileName: file.name,
          fileSize: file.size,
        });
      }
    } finally {
      this.fileUploadInProgress.update(state => ({ ...state, [fieldId]: false }));
    }
  }

  isFieldReadonly(fieldType: FormFieldType): boolean {
    return this.mode() !== 'fill';
  }

  getFieldClass(field: FormFieldConfig): Record<string, boolean> {
    return {
      'form-field': true,
      [`form-field-${field.type}`]: true,
      'form-field-readonly': this.isFieldReadonly(field.type),
    };
  }
}
```

```typescript
import {
  Component, input, inject,
  ChangeDetectionStrategy, OnInit
} from '@angular/core';
import { FormGroup, ReactiveFormsModule } from '@angular/forms';
import { CommonModule } from '@angular/common';
import { MatFormFieldModule } from '@angular/material/form-field';
import { MatInputModule } from '@angular/material/input';
import { MatSelectModule } from '@angular/material/select';
import { MatCheckboxModule } from '@angular/material/checkbox';
import { MatButtonModule } from '@angular/material/button';
import { MatIconModule } from '@angular/material/icon';

@Component({
  selector: 'app-form-field-editor',
  standalone: true,
  imports: [
    CommonModule,
    ReactiveFormsModule,
    MatFormFieldModule,
    MatInputModule,
    MatSelectModule,
    MatCheckboxModule,
    MatButtonModule,
    MatIconModule,
  ],
  templateUrl: './form-field-editor.component.html',
  styleUrls: ['./form-field-editor.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class FormFieldEditorComponent implements OnInit {
  fieldFormGroup = input.required<FormGroup>();
  isLocked = input<boolean>(false);

  FORM_FIELD_TYPES = ['text', 'textarea', 'dropdown', 'number', 'file', 'checkbox'];
  PATTERN_PRESETS = ['email', 'phone', 'url', 'custom'];

  ngOnInit(): void {
    // Initialize type-specific controls as needed
    const form = this.fieldFormGroup();
    if (form) {
      form.get('type')?.valueChanges.subscribe((type) => {
        this.updateTypeSpecificControls(type);
      });
    }
  }

  private updateTypeSpecificControls(type: string): void {
    const form = this.fieldFormGroup();
    if (!form) return;

    // Reset all type-specific controls
    form.patchValue({
      textValidation: form.get('textValidation')?.value || {},
      dropdownOptions: form.get('dropdownOptions')?.value || [],
      numberValidation: form.get('numberValidation')?.value || {},
      fileUploadConfig: form.get('fileUploadConfig')?.value || {},
    });
  }

  onOptionsChange(optionsText: string): void {
    const form = this.fieldFormGroup();
    if (!form) return;

    // Parse options from textarea (one per line or comma-separated)
    const options = optionsText
      .split(/[\n,]/)
      .map(opt => opt.trim())
      .filter(opt => opt.length > 0)
      .map(opt => ({ value: opt.toLowerCase().replace(/\s+/g, '_'), label: opt }));

    form.get('dropdownOptions')?.setValue(options);
  }

  getOptionsText(): string {
    const form = this.fieldFormGroup();
    const options = form?.get('dropdownOptions')?.value || [];
    return options.map((opt: any) => opt.label).join('\n');
  }
}
```
