```typescript
import { Injectable, inject } from '@angular/core';
import { UUID } from '@zerobias-org/types-core-js';
import { Observable, firstValueFrom } from 'rxjs';
import { FormSubmission, FormBuilderConfig } from '../models/form-builder.model';
import { PipelineWriteService } from './pipeline-write.service';
import { GraphQLReadService } from './graphql-read.service';
import { SME_MART_CLASS_IDS } from '../models/sme-mart-class-ids';
import { SME_MART_PIPELINE_ID } from '../constants/pipeline.constants';

@Injectable({ providedIn: 'root' })
export class FormSubmissionService {
  private pipelineWrite = inject(PipelineWriteService);
  private gqlRead = inject(GraphQLReadService);

  /**
   * Create a new form submission (draft) for a project/bid
   */
  async create(projectId: UUID, bidId: UUID): Promise<FormSubmission> {
    if (!projectId || !bidId) {
      throw new Error('projectId and bidId are required to create FormSubmission');
    }

    const now = new Date();
    const submission: FormSubmission = {
      id: crypto.randomUUID(),
      projectId,
      bidId,
      submissionData: {},
      status: 'draft',
      createdAt: now,
      updatedAt: now,
    };

    // Persist via Pipeline
    await this.pipelineWrite.receive(
      SME_MART_PIPELINE_ID,
      [{
        classId: SME_MART_CLASS_IDS.FORM_SUBMISSION,
        object: {
          newFormSubmission: {
            id: submission.id,
            projectId: submission.projectId,
            bidId: submission.bidId,
            submissionData: JSON.stringify(submission.submissionData),
            status: submission.status,
            createdAt: submission.createdAt,
            updatedAt: submission.updatedAt,
          }
        }
      }]
    );

    return submission;
  }

  /**
   * Fetch FormSubmission by id
   */
  async getById(id: UUID): Promise<FormSubmission | null> {
    try {
      const result = await firstValueFrom(
        this.gqlRead.query<{ formSubmission: any }>({
          query: `query GetFormSubmission($id: UUID!) {
            formSubmission(id: $id) {
              id
              projectId
              bidId
              submissionData
              status
              submittedAt
              reviewedAt
              reviewedBy
              createdAt
              updatedAt
            }
          }`,
          variables: { id },
        })
      );

      if (!result?.data?.formSubmission) return null;

      return this.parseFormSubmission(result.data.formSubmission);
    } catch {
      return null;
    }
  }

  /**
   * Get FormSubmission for a specific project/bid pair
   */
  async getByProjectAndBid(projectId: UUID, bidId: UUID): Promise<FormSubmission | null> {
    try {
      const result = await firstValueFrom(
        this.gqlRead.query<{ formSubmissions: { items: any[] } }>({
          query: `query GetFormSubmissionByProjectBid($projectId: UUID!, $bidId: UUID!) {
            formSubmissions(filter: { projectId: $projectId, bidId: $bidId }, pageSize: 1) {
              items {
                id
                projectId
                bidId
                submissionData
                status
                submittedAt
                reviewedAt
                reviewedBy
                createdAt
                updatedAt
              }
            }
          }`,
          variables: { projectId, bidId },
        })
      );

      const item = result?.data?.formSubmissions?.items?.[0];
      if (!item) return null;

      return this.parseFormSubmission(item);
    } catch {
      return null;
    }
  }

  /**
   * Update FormSubmission with new data and status
   */
  async update(id: UUID, updates: Partial<FormSubmission>): Promise<FormSubmission> {
    const payload: any = {
      id,
      updatedAt: new Date(),
    };

    // Map each update field
    if (updates.status !== undefined) payload.status = updates.status;
    if (updates.submittedAt !== undefined) payload.submittedAt = updates.submittedAt;
    if (updates.reviewedAt !== undefined) payload.reviewedAt = updates.reviewedAt;
    if (updates.reviewedBy !== undefined) payload.reviewedBy = updates.reviewedBy;
    if (updates.submissionData !== undefined) {
      payload.submissionData = JSON.stringify(updates.submissionData);
    }

    // Pipeline update call
    await this.pipelineWrite.receive(
      SME_MART_PIPELINE_ID,
      [{
        classId: SME_MART_CLASS_IDS.FORM_SUBMISSION,
        object: payload
      }]
    );

    // Fetch updated entity
    const updated = await this.getById(id);
    if (!updated) {
      throw new Error(`Failed to update FormSubmission ${id}`);
    }
    return updated;
  }

  /**
   * Mark submission as reviewed by a buyer
   */
  async markReviewed(id: UUID, reviewedBy: UUID): Promise<FormSubmission> {
    return this.update(id, {
      status: 'reviewed',
      reviewedAt: new Date(),
      reviewedBy,
    });
  }

  /**
   * Check if submission is in draft state
   */
  isDraft(submission: FormSubmission): boolean {
    return submission.status === 'draft';
  }

  /**
   * Check if form is locked (any submission exists for this project)
   * Returns true if form should be read-only, false if buyer can still edit form config
   * This is the CRITICAL GATE for D-13 (form lock on first submission)
   */
  async getFormSubmissionLock(projectId: UUID): Promise<boolean> {
    try {
      const result = await firstValueFrom(
        this.gqlRead.query<{ formSubmissions: { total: number } }>({
          query: `query CheckFormSubmissions($projectId: UUID!) {
            formSubmissions(filter: { projectId: $projectId }, pageSize: 1) {
              total
            }
          }`,
          variables: { projectId },
        })
      );

      return (result?.data?.formSubmissions?.total || 0) > 0;
    } catch {
      return false;
    }
  }

  /**
   * List all submissions for a project with pagination
   */
  async listByProject(projectId: UUID, pageNumber: number = 1, pageSize: number = 10): Promise<FormSubmission[]> {
    try {
      const result = await firstValueFrom(
        this.gqlRead.query<{ formSubmissions: { items: any[] } }>({
          query: `query ListFormSubmissions($projectId: UUID!, $pageNumber: Int!, $pageSize: Int!) {
            formSubmissions(filter: { projectId: $projectId }, pageNumber: $pageNumber, pageSize: $pageSize) {
              items {
                id
                projectId
                bidId
                submissionData
                status
                submittedAt
                reviewedAt
                reviewedBy
                createdAt
                updatedAt
              }
            }
          }`,
          variables: { projectId, pageNumber, pageSize },
        })
      );

      return (result?.data?.formSubmissions?.items || [])
        .map(item => this.parseFormSubmission(item));
    } catch {
      return [];
    }
  }

  /**
   * Parse FormSubmission from GQL response (convert JSON strings to objects)
   */
  private parseFormSubmission(raw: any): FormSubmission {
    return {
      id: raw.id,
      projectId: raw.projectId,
      bidId: raw.bidId,
      submissionData: typeof raw.submissionData === 'string' 
        ? JSON.parse(raw.submissionData) 
        : (raw.submissionData || {}),
      status: raw.status,
      submittedAt: raw.submittedAt ? new Date(raw.submittedAt) : undefined,
      reviewedAt: raw.reviewedAt ? new Date(raw.reviewedAt) : undefined,
      reviewedBy: raw.reviewedBy,
      createdAt: new Date(raw.createdAt),
      updatedAt: new Date(raw.updatedAt),
    };
  }
}
```
