```typescript
import {
  Component, input, output, signal, inject,
  ChangeDetectionStrategy, OnInit
} from '@angular/core';
import { CommonModule } from '@angular/common';
import { MatSlideToggleModule } from '@angular/material/slide-toggle';
import { MatCardModule } from '@angular/material/card';
import { MatTabsModule } from '@angular/material/tabs';
import { FormBuilderComponent } from '../../../../shared/components/form-builder/form-builder.component';
import { DynamicFormRendererComponent } from '../../../../shared/components/form-builder/dynamic-form-renderer.component';
import { FormSubmissionService } from '../../../../core/services/form-submission.service';
import { RfpWizardService } from '../../../../core/services/rfp-wizard.service';
import { SmeMartProject, FormBuilderConfig } from '../../../../core/models';

@Component({
  selector: 'app-rfp-step-form',
  standalone: true,
  imports: [
    CommonModule,
    MatSlideToggleModule,
    MatCardModule,
    MatTabsModule,
    FormBuilderComponent,
    DynamicFormRendererComponent,
  ],
  templateUrl: './rfp-step-form.component.html',
  styleUrls: ['./rfp-step-form.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class RfpStepFormComponent implements OnInit {
  rfpData = input<SmeMartProject>();
  formConfigSaved = output<FormBuilderConfig>();

  private formSubmissionService = inject(FormSubmissionService);
  private rfpWizardService = inject(RfpWizardService);

  formEnabled = signal(false);
  formConfig = signal<FormBuilderConfig>({ version: 1, fields: [] });
  isFormLocked = signal(false);
  showPreview = signal(false);

  ngOnInit(): void {
    const rfp = this.rfpData();
    if (rfp?.formConfig) {
      const config = typeof rcp.formConfig === 'string' 
        ? JSON.parse(rcp.formConfig) 
        : rcp.formConfig;
      this.formConfig.set(config);
      this.formEnabled.set(true);
    }

    // Check if form is locked (submissions exist)
    this.checkFormLock();
  }

  private async checkFormLock(): Promise<void> {
    const rcp = this.rfpData();
    if (!rcp?.id) return;

    const locked = await this.formSubmissionService.getFormSubmissionLock(rcp.id);
    this.isFormLocked.set(locked);
  }

  onFormEnabledChange(enabled: boolean): void {
    if (!enabled) {
      this.formConfig.set({ version: 1, fields: [] });
    }
    this.formEnabled.set(enabled);
    this.emitChange();
  }

  onFormConfigChange(newConfig: FormBuilderConfig): void {
    this.formConfig.set(newConfig);
    this.emitChange();
  }

  togglePreview(): void {
    this.showPreview.update(v => !v);
  }

  private emitChange(): void {
    if (this.formEnabled()) {
      // Save to RfpWizardState
      this.rfpWizardService.saveStepData('form', this.formConfig());
      this.formConfigSaved.emit(this.formConfig());
    }
  }
}
```

```typescript
async onFormDraftSave(data: Record<string, unknown>): Promise<void> {
  const submission = this.formSubmission();
  if (!submission) return;

  this.isLoadingSubmission.set(true);
  try {
    const updated = await this.formSubmissionService.update(submission.id, {
      submissionData: data,
      status: 'draft',
    });
    this.formSubmission.set(updated);
  } finally {
    this.isLoadingSubmission.set(false);
  }
}

async onFormSubmit(data: Record<string, unknown>): Promise<void> {
  const submission = this.formSubmission();
  if (!submission) return;

  this.isLoadingSubmission.set(true);
  try {
    const updated = await this.formSubmissionService.update(submission.id, {
      submissionData: data,
      status: 'submitted',
      submittedAt: new Date(),
    });
    this.formSubmission.set(updated);
    // Show confirmation message
    this.showSuccessMessage('Form submitted successfully');
  } finally {
    this.isLoadingSubmission.set(false);
  }
}
```

```typescript
import {
  Component, input, signal, inject,
  ChangeDetectionStrategy, OnInit
} from '@angular/core';
import { CommonModule } from '@angular/common';
import { MatCardModule } from '@angular/material/card';
import { MatButtonModule } from '@angular/material/button';
import { MatIconModule } from '@angular/material/icon';
import { DynamicFormRendererComponent } from '../form-builder/dynamic-form-renderer.component';
import { FormSubmissionService } from '../../core/services/form-submission.service';
import { FormSubmission, FormBuilderConfig } from '../../core/models';
import { DatePipe } from '@angular/common';

@Component({
  selector: 'app-bid-review',
  standalone: true,
  imports: [
    CommonModule,
    DatePipe,
    MatCardModule,
    MatButtonModule,
    MatIconModule,
    DynamicFormRendererComponent,
  ],
  templateUrl: './bid-review.component.html',
  styleUrls: ['./bid-review.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class BidReviewComponent implements OnInit {
  formSubmission = input<FormSubmission>();
  formConfig = input<FormBuilderConfig>();
  currentUserId = input<UUID>();

  private formSubmissionService = inject(FormSubmissionService);

  isMarking = signal(false);
  isReviewed = signal(false);

  ngOnInit(): void {
    const submission = this.formSubmission();
    if (submission) {
      this.isReviewed.set(submission.status === 'reviewed');
    }
  }

  async onMarkReviewed(): Promise<void> {
    const submission = this.formSubmission();
    if (!submission || !this.currentUserId()) return;

    this.isMarking.set(true);
    try {
      const updated = await this.formSubmissionService.markReviewed(
        submission.id,
        this.currentUserId()
      );
      this.isReviewed.set(updated.status === 'reviewed');
    } finally {
      this.isMarking.set(false);
    }
  }

  getStatusBadgeColor(status: string): string {
    switch (status) {
      case 'draft':
        return 'accent';
      case 'submitted':
      case 'revised':
        return 'primary';
      case 'reviewed':
        return 'accent';
      default:
        return 'basic';
    }
  }

  getStatusLabel(status: string): string {
    switch (status) {
      case 'draft':
        return 'Draft (not submitted)';
      case 'submitted':
        return 'Submitted';
      case 'revised':
        return 'Revised after initial submission';
      case 'reviewed':
        return 'Reviewed';
      default:
        return status;
    }
  }
}
```
