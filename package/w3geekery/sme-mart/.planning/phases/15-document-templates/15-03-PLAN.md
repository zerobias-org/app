From 15-CONTEXT.md (UI decisions D-05 through D-18):

**Template Library UI (D-05 to D-08):**
- Lives as "Document Templates" tab on `/org` page (consistent with org-level management)
- Tab shows card list of templates with:
  - Template name
  - Type chip (MSA, NDA, SOW, compliance, other)
  - Status badge (Draft, Published, Archived)
  - Variable count
  - Last edited date
  - Actions: Edit (→ `/templates/:id`), Publish, Archive, Delete
- Only Published templates appear in instantiation pickers
- Draft/Archived visible in list but grayed out or with status indicator

**Template Editor Page (D-06):**
- Standalone route `/templates/:id` (not modal)
- Full-screen Milkdown Crepe editor
- Toolbar buttons for:
  - Insert Variable (→ variable picker dropdown)
  - Slash command `/var` (→ inline autocomplete)
  - Preview toggle (switches to rendered HTML with sample data)
  - Save, Publish, Archive buttons
- Side panel on right: Variable management (add/edit/delete custom variables)
  - Built-in vars listed read-only
  - Custom vars: add form (name, label, type, required, defaultValue), edit/delete actions
- Template metadata: name, description, documentType at top

**Instantiation Workflow (D-09 to D-11):**
- Chooser dialog pattern (reusable, scope-aware)
- Dialog shows published templates from org
- Buyer selects one → form auto-populates with custom variable fill-in fields
- Dialog title/button text changes by scope:
  - "Add from Template" (engagement documents)
  - "Add from Template" (project documents)
  - "From Template" (notes panel)
- Same dialog component for all three scopes; context parameter determines scope

**Preview & Variable Insertion (D-12 to D-15):**
- Variable chips in editor: styled inline tokens like Notion mentions
- Requires custom Milkdown plugin to render {{varName}} as chip
- Preview toggle: shows rendered markdown with {{varName}} replaced by sample data
- Built-in vars get realistic fake values; custom vars show label + default or placeholder

**Document Listing Integration (D-16 to D-18):**
- Engagement/Project Documents tab shows mixed list:
  - SmeMartDocument (file uploads)
  - DocumentInstance (template instances)
- Template instances get icon/badge ("Template" chip)
- "Add from Template" button alongside existing "Upload" button
- Clicking template doc shows rendered markdown (expandable or panel)
- Edit button opens Milkdown editor for instance
- Shows "Modified from template" indicator if content differs from original
- Same visibility controls as file uploads (all/buyer_only/provider_only)

**Notes Panel Integration (D-11):**
- "New Note" button becomes split button:
  - Default click = blank note (existing behavior)

```typescript
import { Component, inject, ChangeDetectionStrategy, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormBuilder, FormGroup, ReactiveFormsModule, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { MatButtonModule, MatFormFieldModule, MatInputModule, MatSelectModule, MatToolbarModule, MatIconModule, MatSidenavModule, MatSlideToggleModule } from '@angular/material';
import { toSignal } from '@angular/core';
import { tap, switchMap } from 'rxjs/operators';
import { DocumentTemplateService, VariableSubstitutionService } from '@app/core/services';
import { DocumentTemplate, CustomVariable } from '@app/core/models';
import { MarkdownEditorComponent } from '@app/shared/components/markdown-editor/markdown-editor.component';
import { VariablePanelComponent } from '@app/shared/components/variable-panel/variable-panel.component';

@Component({
  selector: 'app-template-editor',
  standalone: true,
  imports: [
    CommonModule,
    ReactiveFormsModule,
    MatButtonModule,
    MatFormFieldModule,
    MatInputModule,
    MatSelectModule,
    MatToolbarModule,
    MatIconModule,
    MatSidenavModule,
    MatSlideToggleModule,
    MarkdownEditorComponent,
    VariablePanelComponent
  ],
  templateUrl: './template-editor.component.html',
  styleUrl: './template-editor.component.scss',
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class TemplateEditorComponent implements OnInit {
  private route = inject(ActivatedRoute);
  private router = inject(Router);
  private documentTemplateService = inject(DocumentTemplateService);
  private variableSubstitution = inject(VariableSubstitutionService);
  private fb = inject(FormBuilder);

  form!: FormGroup;
  isCreateMode = false;
  previewMode = signal(false);
  customVariables = signal<CustomVariable[]>([]);
  availableVariableNames = signal<string[]>([]);

  template = toSignal(
    this.route.paramMap.pipe(
      switchMap(params => {
        const id = params.get('id');
        if (id === 'new') {
          this.isCreateMode = true;
          return of(null);
        }
        return this.documentTemplateService.getById(id!);
      })
    ),
    { initialValue: null as DocumentTemplate | null }
  );

  ngOnInit() {
    this.initForm();
    this.loadTemplate();
  }

  initForm() {
    this.form = this.fb.group({
      name: ['', [Validators.required, Validators.minLength(3)]],
      description: [''],
      documentType: ['sow', Validators.required],
      content: ['', Validators.required]
    });
  }

  loadTemplate() {
    const template = this.template();
    if (!template) return;

    this.form.patchValue({
      name: template.name,
      description: template.description,
      documentType: template.documentType,
      content: template.content
    });

    const customVars = this.variableSubstitution.parseCustomVariables(template.variableSchema);
    this.customVariables.set(customVars);
    this.updateAvailableVariables();
  }

  updateAvailableVariables() {
    const builtIn = ['buyerOrgName', 'vendorOrgName', 'engagementTitle', 'engagementId', 'projectName', 'projectId', 'effectiveDate', 'expirationDate', 'todayDate'];
    const custom = this.customVariables().map(v => v.name);
    this.availableVariableNames.set([...builtIn, ...custom]);
  }

  onCustomVariablesChanged(variables: CustomVariable[]) {
    this.customVariables.set(variables);
    this.updateAvailableVariables();
  }

  save() {
    if (!this.form.valid) return;

    const dto = {
      ...this.form.value,
      variableSchema: this.customVariables()
    };

    if (this.isCreateMode) {
      const created = await this.documentTemplateService.create({
        ...dto,
        orgId: '' // Get from context
      });
      this.router.navigate(['/templates', created.id]);
    } else {
      const templateId = this.template()?.id;
      if (!templateId) return;
      await this.documentTemplateService.update(templateId, dto);
      await this.loadTemplate();
    }
  }

  async publish() {
    const templateId = this.template()?.id;
    if (!templateId) return;
    await this.documentTemplateService.publish(templateId);
    await this.loadTemplate();
  }

  async confirmArchive() {
    const templateId = this.template()?.id;
    if (!templateId) return;
    const dialogRef = this.dialog.open(ConfirmDialogComponent, {
      data: { title: 'Archive Template', message: 'Archive this template? It will be hidden from template pickers but existing instances are unaffected.' }
    });
    const confirmed = await dialogRef.afterClosed().toPromise();
    if (confirmed) {
      await this.documentTemplateService.archive(templateId);
      this.router.navigate(['/org']);
    }
  }

  goBack() {
    this.router.navigate(['/org']);
  }
}
```

```typescript
import { Component, Inject, OnInit, ChangeDetectionStrategy, signal, inject } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormBuilder, FormGroup, ReactiveFormsModule, Validators } from '@angular/forms';
import { MatDialogModule, MatButtonModule, MatFormFieldModule, MatInputModule, MatSelectModule, MAT_DIALOG_DATA, MatDialogRef } from '@angular/material';
import { DocumentTemplate, CustomVariable } from '@app/core/models';
import { DocumentTemplateService, DocumentInstanceService, VariableSubstitutionService } from '@app/core/services';
import { toSignal } from '@angular/core';

export interface TemplateChooserDialogData {
  scope: 'engagement' | 'project' | 'notes'; // Determines dialog context
  engagementId: string;
  projectId?: string;
  orgId: string;
}

@Component({
  selector: 'app-template-chooser-dialog',
  standalone: true,
  imports: [
    CommonModule,
    ReactiveFormsModule,
    MatDialogModule,
    MatButtonModule,
    MatFormFieldModule,
    MatInputModule,
    MatSelectModule
  ],
  templateUrl: './template-chooser-dialog.component.html',
  styleUrl: './template-chooser-dialog.component.scss',
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class TemplateChooserDialogComponent implements OnInit {
  private documentTemplateService = inject(DocumentTemplateService);
  private documentInstanceService = inject(DocumentInstanceService);
  private variableSubstitution = inject(VariableSubstitutionService);
  private fb = inject(FormBuilder);
  private dialogRef = inject(MatDialogRef<TemplateChooserDialogComponent>);

  readonly data = inject<TemplateChooserDialogData>(MAT_DIALOG_DATA);

  publishedTemplates = signal<DocumentTemplate[]>([]);
  selectedTemplate = signal<DocumentTemplate | null>(null);
  customVariablesForm!: FormGroup;
  step: 'select' | 'fill' = 'select';
  creating = signal(false);

  ngOnInit() {
    // Load published templates for the org
  }

  selectTemplate(template: DocumentTemplate) {
    this.selectedTemplate.set(template);

    // Create form for custom variables
    const customVars = this.variableSubstitution.parseCustomVariables(template.variableSchema);
    const controls: Record<string, any> = {};

    customVars.forEach(v => {
      controls[v.name] = [
        v.defaultValue || '',
        v.required ? Validators.required : []
      ];
    });

    this.customVariablesForm = this.fb.group(controls);

    if (customVars.length > 0) {
      this.step = 'fill'; // Show variable fill form
    } else {
      // No custom vars, instantiate immediately
      this.instantiate();
    }
  }

  async instantiate() {
    const template = this.selectedTemplate();
    if (!template || !this.customVariablesForm) return;

    this.creating.set(true);

    try {
      const dto = {
        templateId: template.id,
        engagementId: this.data.engagementId,
        projectId: this.data.projectId,
        customVariables: this.customVariablesForm.value
      };

      const instance = await this.documentInstanceService.instantiate(dto);
      this.creating.set(false);
      this.dialogRef.close(instance);
    } catch (error) {
      this.creating.set(false);
      console.error('Instantiation failed:', error);
    }
  }

  goBack() {
    this.step = 'select';
    this.selectedTemplate.set(null);
  }

  cancel() {
    this.dialogRef.close();
  }

  getTitleForScope(): string {
    switch (this.data.scope) {
      case 'engagement':
        return 'Add Document from Template';
      case 'project':
        return 'Add Document from Template';
      case 'notes':
        return 'Create Note from Template';
      default:
        return 'Add from Template';
    }
  }
}
```
