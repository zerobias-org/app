    Implement `scripts/seed-zb-provider.ts` as a standalone TypeScript file executable via `ts-node`.

    **File structure:**

    ```typescript
    // Imports
    import { ZerobiasClientApp } from '@zerobias-com/zerobias-client';
    import * as dotenv from 'dotenv';

    // Load .env.local
    dotenv.config({ path: '.env.local' });

    // Constants (from CONTEXT.md § Identifying values)
    const ZEROBIAS_ORG_ID = '57c741cf-a58e-5efc-bf2f-93c4f6cf76ec'; // UAT
    const MPI_CLASS_ID = '7bcf86a5-91dc-520d-b9bf-e308b1078d46';
    const PIPELINE_ID = '43f08afd-7ab9-4e99-a93c-619c46adaabe';
    const CLEANUP_IDS = ['mpi-test-a-cd7105df', 'mpi-test-b-cd7105df'];

    // Seed sections (from COMPANY-INFO-CONVENTION.md, initial cut)
    // At minimum: legal_name, logo_url, short_blurb, long_description, website, years_in_business, employee_count
    // Optional: primary_contact.*, hq_location.*
    interface SeedSection {
      section: string;
      data: string; // All values as string, per MPI shape
    }

    const SEED_SECTIONS: SeedSection[] = [
      { section: 'legal_name', data: 'ZeroBias' },
      { section: 'logo_url', data: 'https://zerobias.com/logo.png' },
      { section: 'short_blurb', data: 'Cybersecurity and compliance automation platform' },
      { section: 'long_description', data: 'ZeroBias is a platform for automating cybersecurity and compliance frameworks...' },
      { section: 'website', data: 'https://zerobias.com' },
      { section: 'years_in_business', data: '10' },
      { section: 'employee_count', data: '201-500' },
      // Optional: primary_contact, hq_location sections can be empty strings or omitted if not seeding them
    ];

    // Helper to build a single MPI record
    function buildMPIRecord(
      section: SeedSection,
      platformProviderTagId?: string
    ): any {
      const id = `mpi-${ZEROBIAS_ORG_ID}-${section.section}`;
      const record: any = {
        id,
        orgId: ZEROBIAS_ORG_ID,
        section: section.section,
        data: section.data,
        status: 'active',
      };

      // Apply distinguisher per Plan 26-01 decision
      // If option (a): populate Object.tag
      if (platformProviderTagId) {
        record.tag = [{ value: platformProviderTagId }];
      }
      // If option (b): add provider_type section (handled by adding it to SEED_SECTIONS)
      // If option (c): no tag needed; Browse Providers filters by orgId in code

      return record;
    }

    // Tag creation (if option a chosen)
    async function ensurePlatformProviderTag(client: ZerobiasClientApp): Promise<string | undefined> {
      // If option (a) chosen in Plan 26-01:
      const tagName = 'sme-mart.provider.platform';

      try {
        // Search for existing tag
        const existing = await client.hydraClient().getTagApi().searchTags({
          name: tagName,
        });

        if (existing.items && existing.items.length > 0) {
          console.log(`[TAG] Found existing tag: ${existing.items[0].id}`);
          return existing.items[0].id;
        }

        // Create tag if not found
        const newTag = await client.hydraClient().getTagApi().createTag({
          name: tagName,
          scope: 'org', // or 'system' — Plan 26-01 decides
          ownerId: ZEROBIAS_ORG_ID,
        });

        console.log(`[TAG] Created new tag: ${newTag.id}`);
        return newTag.id;
      } catch (err) {
        console.error('[TAG] Error (returning undefined, tag is optional):', err);
        return undefined;
      }
      // If option (b) or (c) chosen, return undefined and skip the tag logic
    }

    // Main seed function
    async function seedZBProvider(client: ZerobiasClientApp): Promise<any> {
      console.log('[SEED] Starting ZeroBias provider seed...');

      // Step 1: Get platform-provider tag UUID (if option a)
      let platformProviderTagId: string | undefined;
      // Uncomment only if option (a) chosen in Plan 26-01:
      // platformProviderTagId = await ensurePlatformProviderTag(client);

      // Step 2: Build MPI records
      const records = SEED_SECTIONS
        .filter(s => s.data) // Skip empty fields
        .map(s => buildMPIRecord(s, platformProviderTagId));

      console.log(`[SEED] Built ${records.length} MPI records`);

      // Step 3: Construct payload
      const payload = {
        pipelineId: PIPELINE_ID,
        classId: MPI_CLASS_ID,
        tagIds: [], // Empty — does NOT tag Objects (DECISIONS.md)
        data: records,
        markDeleted: CLEANUP_IDS,
      };

      console.log('[SEED] Payload constructed, calling Pipeline.receive...');
      console.log('[SEED] Data count:', payload.data.length);
      console.log('[SEED] Cleanup ids:', payload.markDeleted);

      // Step 4: Call Pipeline.receive via MCP
      try {
        const result = await client.platformClient().getPipelineApi().receive(payload);
        console.log('[SEED] Success:', result);
        return result;
      } catch (err) {
        console.error('[SEED] ERROR:', err);
        throw err;
      }
    }

    // CLI entry point
    async function main() {
      try {
        // Initialize ZeroBias client with W3Geekery credentials (MCP auth)
        // Expects ZB_ORG_ID, ZB_API_KEY, ZB_TOKEN in .env.local
        const client = await ZerobiasClientApp.getInstance();
        await seedZBProvider(client);
        console.log('[SEED] COMPLETE');
        process.exit(0);
      } catch (err) {
        console.error('[SEED] FATAL:', err);
        process.exit(1);
      }
    }

    main();
    ```
